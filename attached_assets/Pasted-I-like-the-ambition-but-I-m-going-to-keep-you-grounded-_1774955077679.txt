```python
def calculate_score(donor, patient):
    score = 0

    # 1. Blood compatibility
    if donor["blood_group"] == patient["blood_group"]:
        score += 50
    elif is_compatible(donor, patient):
        score += 30

    # 2. HLA match (if exists)
    if donor.get("hla") and patient.get("hla"):
        hla_score = compare_hla(donor["hla"], patient["hla"])
        score += hla_score * 20

    # 3. Distance priority
    distance = get_distance(donor, patient)

    if distance < 5:
        score += 30
    elif distance < 50:
        score += 20
    elif distance < 200:
        score += 10

    # 4. Reliability
    score += donor["reliability_score"] * 0.2

    # 5. Availability
    if donor["available"]:
        score += 20

    return score
```
